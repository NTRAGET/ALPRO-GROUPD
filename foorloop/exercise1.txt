using System;

class Program
{
    static void Main()
    {
        Console.WriteLine("=== FOR LOOP ===");
        for (int i = 1; i <= 30; i++)
        {
            if (i % 2 == 0)
                Console.WriteLine($"{i} adalah angka genap");
            else
                Console.WriteLine($"{i} adalah angka ganjil");
        }
    }
}
